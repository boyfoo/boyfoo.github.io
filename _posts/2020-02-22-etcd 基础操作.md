### docker-compose 创建etcd

```yaml
version: "3"

services:

  etcd:
    image: quay.io/coreos/etcd:v3.3.12
    ports:
    - 2379:2379
    - 2380:2380
    environment:
      ETCDCTL_API: 3
    #     TZ: CST-8
    #     LANG: zh_CN.UTF-8
    command:
    - "/usr/local/bin/etcd"
    - "--debug"
    - "--name"
    - "s1"
    - "--data-dir"
    - "/etcd-data"
    - "--advertise-client-urls"
    - "http://0.0.0.0:2379"
    - "--listen-client-urls"
    - "http://0.0.0.0:2379"
    - "--initial-advertise-peer-urls"
    - "http://0.0.0.0:2380"
    - "--listen-peer-urls"
    - "http://0.0.0.0:2380"
    - "--initial-cluster-token"
    - "tkn"
    - "--initial-cluster"
    - "s1=http://0.0.0.0:2380"
    - "--initial-cluster-state"
    - "new"
```

### etcd 命令行操作

```bash
# 设置
$ etcdctl put "name" "zx"
# 获取
$ etcdctl get "name"

# 目录似路径
$ etcdctl put "/cron/jobs/job1" "asdasd"
$ etcdctl put "/cron/jobs/job2" "5wegxc23"

# 按照前缀获取
$ etcdctl get "/cron/jobs/" --prefix

# 监听路径
$ etcdctl watch "/cron/jobs/" --prefix
```
